const searchForm = document.getElementById('search-form'); const searchInput = document.getElementById('search-input'); const searchButton = document.getElementById('search-button'); searchForm.addEventListener('submit', event => { event.preventDefault(); const query = searchInput.value.trim(); if (query) { const searchEngines = [ { name: "Neeva", url: `https://neeva.com/search?q=${encodeURIComponent(query)}`, accuracy: 0.9 }, { name: "Perplexity AI", url: `https://www.perplexity.ai/search?q=${encodeURIComponent(query)}`, accuracy: 0.8 }, { name: "Andi", url: `https://andisearch.com/?query=${encodeURIComponent(query)}`, accuracy: 0.7 }, { name: "You.com", url: `https://you.com/search?q=${encodeURIComponent(query)}&fromSearchBar=true&tbm=youchat`, accuracy: 0.6 }, { name: "Phind", url: `https://phind.com/search?q=${encodeURIComponent(query)}`, accuracy: 0.5 }, { name: "Bing", url: `https://bing.com/search?q=${encodeURIComponent(query)}`, accuracy: 0.4 } ]; // Sort search engines by accuracy in ascending order searchEngines.sort((a, b) => a.accuracy - b.accuracy); // Open the search engines in the opposite order of accuracy for (let i = searchEngines.length - 1; i >= 0; i--) { window.open(searchEngines[i].url, "_blank"); } } });
